def get_tokens(doc):
    return [tok.orth_ for tok in doc]


def get_iob_tags(doc):
    return [
        "O" if tok.ent_iob_ == "O" else f"{tok.ent_iob_}-{tok.ent_type_}" for tok in doc
    ]

import spacy
from spacy.tokens import DocBin
from spacy.training import Example
from nervaluate import Evaluator

# Step 1: Data Preparation
# Assume you have training, validation, and test data in the spaCy format (DocBin)
train_data_path = "./train.spacy"
valid_data_path = "./valid.spacy"
test_data_path = "./test.spacy"

# Load spaCy model
nlp = spacy.blank("en")

# Load training, validation, and test data
train_docs = list(DocBin().from_disk(train_data_path).get_docs(nlp.vocab))
valid_docs = list(DocBin().from_disk(valid_data_path).get_docs(nlp.vocab))
test_docs = list(DocBin().from_disk(test_data_path).get_docs(nlp.vocab))

# Step 2: Model Training
# Initialize NER component
ner = nlp.add_pipe("ner")

# Prepare training examples
train_examples = []
for doc in train_docs:
    ents = [(ent.start_char, ent.end_char, ent.label_) for ent in doc.ents]
    train_examples.append(Example.from_dict(doc, {"entities": ents}))

# Train the NER pipeline
nlp.begin_training()
for epoch in range(10):  # Adjust number of epochs as needed
    for example in train_examples:
        nlp.update([example], drop=0.5)  # Adjust dropout rate as needed

# Step 3: Evaluation
# Prepare predictions
pred_docs = list(nlp.pipe(test_docs))

# Prepare gold annotations
gold_annotations = []
for doc in test_docs:
    gold_annotations.append([(ent.start_char, ent.end_char, ent.label_) for ent in doc.ents])

# Evaluate using nervaluate
evaluator = Evaluator(gold_annotations, pred_docs, tags=["LOC", "PER", "ORG", "MISC"], loader="list")
scores = evaluator.evaluate()

# Print evaluation results
print("Evaluation Results:")
for ent_type, metrics in scores.items():
    print(f"Entity Type: {ent_type}")
    print(metrics)
